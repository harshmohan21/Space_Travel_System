import java.util.Arrays;
import java.util.List;

public class BatchConfigValidator {

    private static final List<String> VALID_DATA_TYPES = Arrays.asList(
            "Integer", "Double", "Boolean", "Date", "Time", "Timestamp", "String"
    );

    public static void validateDataTypes(List<String> dataTypes) {
        for (String dataType : dataTypes) {
            if (!VALID_DATA_TYPES.contains(dataType)) {
                throw new IllegalArgumentException("Invalid data type: " + dataType);
            }
        }
    }
}
