private Object readValue(FieldSet fieldSet, String column) {
    // Try to read as integer
    try {
        return fieldSet.readInt(column);
    } catch (Exception ignored) {}

    // Try to read as double
    try {
        return fieldSet.readDouble(column);
    } catch (Exception ignored) {}

    // Try to read as boolean only if the value is exactly "true" or "false"
    String value = fieldSet.readString(column);
    if ("true".equalsIgnoreCase(value) || "false".equalsIgnoreCase(value)) {
        return Boolean.parseBoolean(value);
    }

    // Return as string
    return value;
}
