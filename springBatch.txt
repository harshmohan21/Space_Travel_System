import static org.junit.jupiter.api.Assertions.assertEquals;
import org.junit.jupiter.api.Test;

public class UnescapeStringTest {

    @Test
    public void testUnescapeString_withTab() {
        String input = "Hello\\tWorld";
        String expected = "Hello\tWorld";
        String actual = UnescapeString.unescapeString(input);
        assertEquals(expected, actual);
    }

    @Test
    public void testUnescapeString_withNewline() {
        String input = "Hello\\nWorld";
        String expected = "Hello\nWorld";
        String actual = UnescapeString.unescapeString(input);
        assertEquals(expected, actual);
    }

    @Test
    public void testUnescapeString_withCarriageReturn() {
        String input = "Hello\\rWorld";
        String expected = "Hello\rWorld";
        String actual = UnescapeString.unescapeString(input);
        assertEquals(expected, actual);
    }

    @Test
    public void testUnescapeString_withBackspace() {
        String input = "Hello\\bWorld";
        String expected = "Hello\bWorld";
        String actual = UnescapeString.unescapeString(input);
        assertEquals(expected, actual);
    }

    @Test
    public void testUnescapeString_withFormFeed() {
        String input = "Hello\\fWorld";
        String expected = "Hello\fWorld";
        String actual = UnescapeString.unescapeString(input);
        assertEquals(expected, actual);
    }

    @Test
    public void testUnescapeString_withBackslash() {
        String input = "Hello\\\\World";
        String expected = "Hello\\World";
        String actual = UnescapeString.unescapeString(input);
        assertEquals(expected, actual);
    }

    @Test
    public void testUnescapeString_withDoubleQuote() {
        String input = "Hello\\\"World";
        String expected = "Hello\"World";
        String actual = UnescapeString.unescapeString(input);
        assertEquals(expected, actual);
    }

    @Test
    public void testUnescapeString_withSingleQuote() {
        String input = "Hello\\'World";
        String expected = "Hello'World";
        String actual = UnescapeString.unescapeString(input);
        assertEquals(expected, actual);
    }

    @Test
    public void testUnescapeString_withMixedEscapes() {
        String input = "Hello\\tWorld\\nNew\\rLine\\bBack\\fForm\\\"Quote\\'Single\\\\" + "End";
        String expected = "Hello\tWorld\nNew\rLine\bBack\fForm\"Quote'Single\\End";
        String actual = UnescapeString.unescapeString(input);
        assertEquals(expected, actual);
    }

    @Test
    public void testUnescapeString_withoutEscapes() {
        String input = "Hello World";
        String expected = "Hello World";
        String actual = UnescapeString.unescapeString(input);
        assertEquals(expected, actual);
    }
}
